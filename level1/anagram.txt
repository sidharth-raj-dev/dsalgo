def isAnagram(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: bool
        """
        map1 = {}
        map2 = {}

        for i,val in enumerate(s):
            if val in map1:
                map1[val] += 1
            else:
                map1[val] = 0

        for i,val in enumerate(t):
            if val in map2:
                map2[val] += 1
            else:
                map2[val] = 0
        
        return map1 == map2
